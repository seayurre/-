#결측값(빈칸)을 채우고 최대값 찾기
#결측값은 그냥 -999로 두고 최댓값 구하기

import csv
max_temp= -999 #최대값 비교하기 위한 초기값 (-INF)
max_data=''    # 최고기온의 날짜를 저장할 변수

f=open(path, encoding='cp949')
data=csv.reader(f)
header = next(data)

for row in data:
  if row[-1]=='': #최고기온이 없다면, 빈칸이라면,
    row[-1]=-999  # -999로 채워줌, 빈칸이었음을 표시
  row[-1] = float(row[-1])
  if max_temp < row[-1]: #최대값을 가진 날짜와 최고 기온 추출
    max_data = row[0]
    max_temp = row[-1]
f.close()
print(max_data, max_temp)
